   extension configuration
      [glname] -> false -- poisoned extension
      [glname] -> true -- allowed unchecked extension
      [glname] -> nil -- checked only extension - extension name as check string
      [glname] -> 'asdf' -- checked only extension - 'asdf' as check string
      [glname] -> { 'asdf', 'bbbb' } -- checked only extension - allowed in 'asdf' or 'bbbb' check blocks

   symbol configuration
      [glname] -> false -- poisoned symbol
      [glname] -> _G.ignored -- ignore symbol usages entirely
      [glname] -> nil or true -- continue processing
